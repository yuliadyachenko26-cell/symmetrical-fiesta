        let currentCard = null;
        let orientation = "upright";

        function drawCard() {
            currentCard = deck[Math.floor(Math.random() * deck.length)];
            orientation = Math.random() < 0.5 ? "reversed" : "upright";
            document.getElementById("card").classList.remove("flipped");
            document.getElementById("interpretation").innerHTML = "<p>Flipping your card...</p>";
            document.getElementById("back").style.backgroundImage = "none";

            const img = new Image();
            img.onload = () => {
                setTimeout(() => {
                    document.getElementById("back").style.backgroundImage = `url(${currentCard.img})`;
                    document.getElementById("card").classList.add("flipped");
                    showMeaning();
                }, 300);
            };
            img.onerror = () => {
                document.getElementById("back").style.backgroundImage = "linear-gradient(45deg, #8b4513, #d4af37)";
            };
            img.src = currentCard.img;
        }

        function showMeaning() {
            const meaning = orientation === "reversed" ? currentCard.reversed : currentCard.upright;
            document.getElementById("interpretation").innerHTML = `
                <h2>${currentCard.name} <em>(${orientation})</em></h2>
                <p><strong>${meaning}</strong></p>
                <p><small>Share this reading: <a href="?c=${encodeURIComponent(currentCard.name)}&o=${orientation}" style="color:#d4af37">Copy link</a></small></p>
            `;
        }

        function saveEntry() {
            if (!currentCard) return alert("Draw a card first!");
            const note = document.getElementById("note").value.trim();
            const entries = JSON.parse(localStorage.getItem("tarotJournal") || "[]");
            entries.push({
                date: new Date().toLocaleDateString(),
                card: currentCard.name,
                orientation,
                meaning: orientation === "reversed" ? currentCard.reversed : currentCard.upright,
                note
            });
            localStorage.setItem("tarotJournal", JSON.stringify(entries));
            document.getElementById("note").value = "";
            alert("Saved to your journal!");
        }

        function toggleEntries() {
            const entries = JSON.parse(localStorage.getItem("tarotJournal") || "[]");
            const div = document.getElementById("entries");
            if (entries.length === 0) {
                div.innerHTML = "<p>No entries yet. Start reflecting!</p>";
                return;
            }
            div.innerHTML = "<h3>Past Readings</h3>" + entries.reverse().map(e => `
                <div class="entry">
                    <strong>${e.date} • ${e.card} (${e.orientation})</strong><br>
                    ${e.meaning}<br>
                    ${e.note ? "<em>\\"" + e.note + "\\"</em>" : ""}
                </div>
            `).join("");
        }

        // Keyboard support
        document.addEventListener("keydown", e => e.key === "Enter" && drawCard());

        // Share link support
        const params = new URLSearchParams(location.search);
        if (params.has("c")) {
            setTimeout(drawCard, 800);
        }
